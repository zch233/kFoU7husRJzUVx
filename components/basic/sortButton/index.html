        # SortButton 排序按钮

        # 基础用法

        <template>
  <div>
    <GupoSortButton :data="singleSortButtonData1" @change="handleChange" />
    <GupoSortButton :data="singleSortButtonData2" @change="handleChange" />
  </div>
</template>
<script>
export default {
  data() {
    return {
      singleSortButtonData1: {
        icon: 'visits',
        label: '访问量',
        type: 'view_time',
      },
      singleSortButtonData2: {
        label: '访问量',
        type: 'view_time',
      }
    }
  },
  methods: {
    handleChange(e) {
      alert(JSON.stringify(e))
    }
  }
};
</script>

        # 多个排序按钮

        默认排序

        <template>
  <div>
    <GupoSortButtonGroup :data="sortButtonData" />
    <h4>默认排序</h4>
    <GupoSortButtonGroup :data="sortButtonData" defaultSelected="update_time" defaultSort="ascend" />
  </div>
</template>
<script>
export default {
  data() {
    return {
      sortButtonData: [
        {
          icon: 'visits',
          label: '访问量',
          type: 'view_time',
        },
        {
          icon: 'updateTime',
          label: '更新时间',
          type: 'update_time',
        },
      ],
    }
  }
};
</script>

        # SortButton Attributes

        参数 说明 类型 可选值 默认值
        data { icon: 图标名称, label: 按钮文字, type: 排序数据 } object - —
        selected 默认是否选中 boolean - —
        defaultSort 默认排序值 string 'ascend', 'descend' 'descend'

        # SortButtonGroup Attributes

        参数 说明 类型 可选值 默认值
        data { icon: 图标名称, label: 按钮文字, type: 排序数据 }[] array - —
        defaultSelected 默认选中哪一个 string - —
        defaultSort 默认排序值 string 'ascend', 'descend' 'descend'

        # Event

        事件名称 说明 回调参数
        change 在排序改变时触发 sortBy: 选中数据的 type, sort: 'descend' 或 'descend'
