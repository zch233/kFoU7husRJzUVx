        # LabelSelector 标签选择器

        # 基础用法

        已选择标签：Java
        <template>
  <GupoLabelSelector v-model="selected" :options="options">
    <div>已选择标签：{{ selected.join('、') }}</div>
  </GupoLabelSelector>
</template>
<script>
export default {
  data() {
    return {
      selected: ['Java'],
      options: ['Java', 'JavaScript', 'PHP', 'Python', 'CPP', 'Go'],
    }
  }
};
</script>

        # 使用 hover 激活

        已选择标签：Java
        <template>
  <GupoLabelSelector v-model="triggerSelected" :options="options" trigger="hover">
    <div>已选择标签：{{ triggerSelected.join('、') }}</div>
  </GupoLabelSelector>
</template>
<script>
export default {
  data() {
    return {
      triggerSelected: ['Java'],
      options: ['Java', 'JavaScript', 'PHP', 'Python', 'CPP', 'Go'],
    }
  }
};
</script>

        # 自定义菜单弹出位置

        已选择标签：Java
        <template>
  <GupoLabelSelector v-model="placementSelected" :options="options" placement="bottom-end">
    <div>已选择标签：{{ placementSelected.join('、') }}</div>
  </GupoLabelSelector>
</template>
<script>
export default {
  data() {
    return {
      placementSelected: ['Java'],
      options: ['Java', 'JavaScript', 'PHP', 'Python', 'CPP', 'Go'],
    }
  }
};
</script>

        参数 说明 类型 可选值 默认值

        options 选项数据源 Array[string] - —
        title 标题 String - 添加标签
        trigger 触发下拉的行为 String hover, click click
        placement 菜单弹出位置 String (top,bottom,left,right)(-start,-end) bottom-start
